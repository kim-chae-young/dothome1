parallax03

scrollTop과 OffsetTop을 이용하여 패럴랙스 스크롤링 효과를 완성하였습니다.

jquery

let scrollTop = $(window).scrollTop();
function scroll(){
    const newScrollTop = $(window).scrollTop();
    const different = newScrollTop - scrollTop;

    console.log("scrollTop :" + scrollTop);
    console.log("newScrollTop :" + newScrollTop);
    console.log("different :" + different);

    // $(".content-item-imgWrap").css({"transform":"rotate("+different +"deg)"});
    gsap.to(".content-item-imgWrap", {duration:.3, skewY: different * 0.1, ease:Power2.ease})
    gsap.to(".content-item-num", {duration:.6, x: different * 0.3, ease:Power1.ease})
    gsap.to(".content-item-desc", {duration:.6, skewX: different * 0.2, ease:Power4.ease})

    scrollTop = newScrollTop;
    requestAnimationFrame(scroll);
}
scroll();
