    requestAnimationFrame(scroll);
}

let lastScrollTop = 0;
const totalWidth = $("#contents").width();
$("body").css("height", totalWidth);

$(window).scroll(function(){
    let wScroll = $(window).scrollTop();
    let wWidth = $(window).width();
    let wHeight = $(window).height();
    let dHeight = $("body").height() - wHeight; //전체세로 값 - 현재화면 세로 값
    let dWidth = totalWidth - wWidth; //전체 가로 값 - 현재 화면 가로 값
    let step = dWidth/dHeight;
    let goLeft = wScroll * step;
    // console.log(goLeft);
    $("#contents").css("left", -goLeft);

    clearTimeout($.data(this,"scrollCheck"));

    $.data(this, "scrollCheck", setTimeout(function(){
        $(".character").removeClass("walking")
    },100))
    
    if( wScroll > lastScrollTop ) {
        $(".character").removeClass("left");
        $(".character").addClass("walking");
    } else {
        $(".character").addClass("left");
        // $(".character").removeClass("left")
    }
    lastScrollTop = wScroll;

    if(wScroll > $("#section1").offset().left + $(window).width()){
        gsap.to("#section1 .imgWrap", {duration: 0.4, opacity:1, top:"50%"});
        gsap.to("#section1 .title", {duration: 0.4, opacity:1, delay: 0.4, top:"40%"})

function scrollReveal(){
const reveal = document.querySelectorAll(".reveal");
reveal.forEach(elem => {
    let scrollTop = document.documentElement.scrollTop +window.innerHeight/2;
    let offsetTop = elem.offsetTop + elem.parentElement.offsetTop;
    // console.log(el.offsetTop);

    if(scrollTop > offsetTop){
        elem.classList.add("show");
    }
})
